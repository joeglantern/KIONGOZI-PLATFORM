### Automatic Media Detection

```typescript
// In the markdown renderer (img component)
img: ({ node, src, alt, ...props }: any) => {
  // Check if this is video or audio
  const isVideo =
    alt?.toLowerCase().includes('video') ||
    src?.match(/\.(mp4|webm|ogg|mov)$/i);

  const isAudio =
    alt?.toLowerCase().includes('audio') ||
    src?.match(/\.(mp3|wav|ogg|m4a)$/i);

  // Render MediaPlayer for video/audio
  if (isVideo || isAudio) {
    return (
      <MediaPlayer
        type={isVideo ? 'video' : 'audio'}
        src={src}
        title={alt}
        transcript={/* extracted from content */}
        subtitles={/* if available */}
      />
    );
  }

  // Regular image
  return <img src={src} alt={alt} />;
}
```
